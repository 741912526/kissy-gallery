1 /**
  2  *  文件拖拽插件
  3  *  @author 飞绿
  4  */
  5 KISSY.add('gallery/form/1.1/uploader/plugins/filedrop/filedrop', function(S, Node, Base) {
  6     var EMPTY = '',
  7         $ = Node.all,
  8         UA = S.UA;
  9 
 10     var FileDrop = function(config) {
 11         var self = this;
 12         // console.log(config);
 13         FileDrop.superclass.constructor.call(self, config);
 14 //        console.log($(config.target), self.get('target'));
 15         self.set('mode', getMode());
 16     };
 17 
 18     var getMode = function() {
 19         if(UA.webkit >= 7 || UA.firefox >= 3.6 ) {
 20             return 'supportDrop';
 21         }
 22         if(UQ.ie) {
 23             return 'notSupportDropIe';
 24         }
 25         if(UA.webkit < 7 || UA.firefox < 3.6) {
 26             return 'notSupportDrop';
 27         }
 28     };
 29     
 30     S.mix(FileDrop, {
 31         event: {
 32             'AFTER_DROP': 'afterdrop'       
 33         }
 34     });
 35 
 36     S.extend(FileDrop, Base, {
 37         render: function() {
 38 //            console.log('render', this.get('target'));        
 39             var self = this;
 40             self._createDropArea();
 41             console.log('after randeer');
 42             // self.fire('afterRender', {'buttonWrap': self.get('buttonWrap'), 'config': {'tpl' : self.get('btnTpl')}});
 43             self.fire('afterRender', {'buttonTarget': self.get('buttonWrap')});
 44         },
 45         /**
 46          * 显示拖拽区域
 47          */
 48         show: function() {
 49             var self = this,
 50                 dropContainer = self.get('dropContainer');
 51             dropContainer.show();
 52         },
 53         /**
 54          * 隐藏拖拽区域
 55          */
 56         hide: function() {
 57             var self = this,
 58                 dropContainer = self.get('dropContainer');
 59             dropContainer.hide();
 60         },
 61         /**
 62          * ?
 63          */
 64         reset: function() {},
 65         /**
 66          * 创建拖拽区域
 67          */
 68         _createDropArea: function() {
 69             var self = this,
 70                 target = $(self.get('target')),
 71                 mode = self.get('mode'),
 72                 html = S.substitute(self.get('tpl')[mode], {name: self.get('name')}),
 73                 dropContainer = $(html), 
 74                 buttonWrap = dropContainer.all('.J_ButtonWrap');
 75             // console.log(buttonWrap);
 76             dropContainer.insertAfter(target);
 77             dropContainer.on('dragover', function(ev) {
 78                 ev.stopPropagation();
 79                 ev.preventDefault();
 80             });
 81             dropContainer.on('drop', function(ev) {
 82                 ev.stopPropagation();
 83                 ev.preventDefault();
 84                 self._dropHandler(ev);
 85             });
 86             self.set('dropContainer', dropContainer);
 87             self.set('buttonWrap', buttonWrap);
 88         },
 89         /**
 90          * 处理拖拽时间 
 91          */
 92         _dropHandler: function(ev) {
 93             console.log('handler');
 94             var self = this,
 95                 event = FileDrop.event,
 96                 fileList = ev.originalEvent.dataTransfer.files,
 97                 files = [];
 98 //            console.log(fileList.length);
 99             if(fileList.lenght == 0) {return;}
100 
101             S.each(fileList, function(f) {
102                 if(S.isObject(f)) {
103                     files.push({'name' : f.name, 'type': f.type, 'size': f.size, 'input': {files:[f]}});
104                 }
105             });
106             self.fire(event.AFTER_DROP, {files: files});
107         },
108         _setDisabled: function() {}
109     }, {
110         ATTRS: {
111             target: {
112                 value: null        
113             },       
114             fileInput: {
115                 value: EMPTY           
116             },
117             dropContainer: {
118                 value: EMPTY               
119             },
120             tpl: {
121                 value: {
122                     supportDrop: '<div class="drop-wrapper">' + 
123                                     '<p>直接拖拽图片到这里，</p>' + 
124                                     '<p class="J_ButtonWrap">或者' + 
125                                     '</p>' +
126                                 '</div>',
127                     notSupportDropIe: '<div class="drop-wrapper">' + 
128                                         '<p>您的浏览器只支持传统的图片上传，</p>' + 
129                                         '<p class="suggest J_ButtonWrap">推荐使用chrome浏览器或firefox浏览器' + 
130                                         '</p>' + 
131                                     '</div>',
132                     notSupportDrop: '<div class="drop-wrapper">' + 
133                                         '<p>您的浏览器只支持传统的图片上传，</p>' + 
134                                         '<p class="suggest J_ButtonWrap">推荐升级您的浏览器' + 
135                                         '</p>' +
136                                     '</div>'
137                 }
138             },
139             btnTpl: {
140                 value: '<input type="file" name="dropFile" hidefoucs="true" class="file-input" />'
141             },
142             name: {
143                 value: '',
144                 setter: function(v) {
145                 }
146             },
147             disabled: {
148                 value: false,
149                 setter: function(v) {
150                     this._setDisabled(v);
151                     return v;
152                 }
153             },
154             cls: {
155                 disabled: 'drop-area-disabled'
156             }
157         }
158     });
159 
160     return FileDrop;
161 }, {requires: ['node', 'base']});
