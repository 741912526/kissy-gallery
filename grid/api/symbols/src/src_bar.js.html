 82 		_bindUI: function() {
 83 			var self = this,
 84 				attrs = self.__attrs,
 85 				attr, m;
 86 
 87 			for (attr in attrs) {
 88 				if (attrs.hasOwnProperty(attr)) {
 89 					self.on('after' + capitalFirst(attr) + 'Change', function(ev) {
 90 							self[attr] = ev.newVal;
 91 					});
 92 					m = UI_SET + capitalFirst(attr);
 93 					if (self[m]) {
 94 						// 自动绑定事件到对应函数
 95 						(function(attr, m) {
 96 							self.on('after' + capitalFirst(attr) + 'Change', function(ev) {
 97 								self[m](ev.newVal, ev);
 98 							});
 99 						})(attr, m);
100 					}
101 				}
102 			}

370 		_afterStoreLoad : function (store, params) {
371 			var _self = this,
372 				pageSize = _self.get('pageSize'),
373 				start = 0,  //页面的起始记录
374 				end = 0,    //页面的结束记录

379                 start = params.start || 0;

381                 start  = 0;
382             }
383             //设置加载数据后翻页栏的状态
384             totalCount = store.getTotalCount();
385             end = totalCount - start > pageSize ? start + store.getCount() : totalCount;
386             totalPage = parseInt((totalCount + pageSize - 1) / pageSize, 10);
387             totalPage = totalPage > 0 ? totalPage : 1;
388             curPage = parseInt(start / pageSize, 10) + 1;
389 
390 			_self.start = start;
391 			_self.end = end;
392 			_self.totalCount = totalCount;
393 			_self.curPage = curPage;
394 			_self.totalPage = totalPage;

398             _self._setNumberPages();

495 			var _self = this;
496 			return value && value > 0 && value <= _self.totalPage && value !== _self.curPage;
497 		},
498 		//分页栏上的按钮事件
499 		_pageAction : function (btnEl) {
500 			var _self = this,
501 				page = 1;
502 			if (btnEl.hasClass(CLS_FIRST)) {
503 				page = 1;
504 			} else if (btnEl.hasClass(CLS_NEXT)) {
505 				page = _self.curPage + 1;
506 			} else if (btnEl.hasClass(CLS_PREV)) {
507 				page = _self.curPage - 1;
508 			} else if (btnEl.hasClass(CLS_LAST)) {
509 				page = _self.totalPage;
510 			} else if (btnEl.hasClass(CLS_OK)) {
511 				var bar = _self.get('bar');
512 					inputEl = bar.one('.' + CLS_ITEM_INPUT),
513 					value = parseInt(inputEl.val(), 10);
514 				if(_self._isPageAllowRedirect(value)){
515 					page = value;
516 				}else{
517 					inputEl.val(_self.curPage);
518 					return;
519 				}
520 			}
521 			_self.skipToPage(page);
522 
523 		},
524 		/**
525 		* 跳到指定页
526 		* @param {Number} page 目标页
527 		*/
528 		skipToPage : function (page) {
529 			var _self = this,
530 				store = _self.get('store'),
531 				pageSize = _self.get('pageSize'),
532 				index = page - 1,
533 				start = index * pageSize;
534 			var reslut = _self.fire('beforepagechange', {from : _self.curPage, to : page});
535 			if(store && reslut !== false){
536 				store.load({ start : start, limit : pageSize, pageIndex : index });
537 			}
538 		},
539 		//设置按钮状态
540 		_setAllButtonsState : function () {
541 			var _self = this,
542 				store = _self.get('store'),
543 				items = _self.get('items');
544 			
545 			/**
546 			* @private
547 			*/
548 			function enableButtons(btnNames, enable) {
549 				S.each(items, function (item) {
550 					if(S.inArray(item.id,btnNames)){
551 						item.set('disabled',!enable);
552 					}
553 				});
554 			}
555 			if (store) {
556                 enableButtons(['btnpre', 'btnnext', 'btnfirst', 'btnlast', 'btnok'], true);
557             }
558 
559             if (_self.curPage === 1) {
560                 enableButtons(['btnpre', 'btnfirst'], false);
561             }
562             if (_self.curPage === _self.totalPage) {
563                 enableButtons(['btnnext', 'btnlast'], false);
564             }
565 		},
566 		//显示页码和总页数

568 			var _self = this,

570 				input = bar.one('.' + CLS_ITEM_INPUT),

572 				totalEl = bar.one('.' + CLS_TEXT_COUNT);
573 			if (input && text) {
574 				input.val(_self.curPage);

576 			}
577 			if (totalEl) {
578 				totalEl.text(_self.totalCount);
579 			}
580 		}
